        <script>
        const pages = ['page-home', 'page-stroud', 'page-clarity', 'page-mtbr', 'page-ochre', 'page-pathway'];
        const navLinks = ['nav-home', 'nav-mission', 'nav-leadership', 'nav-portfolio', 'nav-contact'];

        function updateActiveNav(activeId) {
            navLinks.forEach(id => {
                const link = document.getElementById(id);
                if (link) {
                    link.classList.remove('active-link');
                }
            });
            if (activeId && document.getElementById(`nav-${activeId}`)) {
                document.getElementById(`nav-${activeId}`).classList.add('active-link');
            } else if (activeId === 'home') {
                document.getElementById('nav-home').classList.add('active-link');
            }
        }

        function showPage(pageId, scrollToSection = null) {
            // 1. Handle Visibility
            pages.forEach(id => {
                const page = document.getElementById(id);
                if (page) {
                    page.classList.add('hidden');
                    page.classList.remove('fade-in');
                }
            });

            const targetPage = document.getElementById(pageId);
            if (targetPage) {
                targetPage.classList.remove('hidden');
                targetPage.classList.add('fade-in');
            }

            // 2. Handle Navigation Highlighting and Scrolling
            if (pageId === 'page-home') {
                updateActiveNav(scrollToSection || 'home');
                if (scrollToSection) {
                    document.getElementById(scrollToSection).scrollIntoView({ behavior: 'smooth', block: 'start' });
                } else {
                    window.scrollTo({ top: 0, behavior: 'smooth' });
                }
            } else {
                // If showing a subsidiary page, remove all active nav states and scroll to top
                updateActiveNav(null);
                window.scrollTo({ top: 0, behavior: 'smooth' });
            }
        }
